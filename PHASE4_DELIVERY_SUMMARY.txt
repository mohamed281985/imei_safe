💡 SIMPLE USAGE
───────────────────────────────────────────────────────────────

Before (Manual):
  import axios from 'axios'
  const res = await axios.post('/api/user', data)
  ❌ No CSRF protection

After (Automatic):
  import { useApiCall } from '@/hooks/useApiCall'
  const { data, execute } = useApiCall()
  await execute('/api/user', 'post', data)
  ✅ Full CSRF protection automatic!
